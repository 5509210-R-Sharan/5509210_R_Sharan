def num_pairs(n, s):
    pairs = 0
    checked = []
    for i in range(len(s)):
        if s[i] not in checked:
            num = s.count(s[i])
            pairs = pairs + num//2
            checked.append(s[i])
    return pairs
    
n = int(input())
s = list(map(int, input().split()))
print(num_pairs(n, s))
