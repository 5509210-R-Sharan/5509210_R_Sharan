12. Week-2 Counter game Problem

def counter_game(n):
    m = 0
    while n > 1:
        if n & (n - 1) == 0:
            n //= 2
        else:
            p = 1
            while p * 2 < n:
                p *= 2
            n -= p
        m += 1
    
    if m % 2 == 1:
        return "Louise"
    else:
        return "Richard"
    
t = int(input().strip())
for i in range(t):
    n = int(input().strip())
    print(counter_game(n))
