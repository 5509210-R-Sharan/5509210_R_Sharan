2. Week-3 New Year Chaos Problem

def new_year_chaos(q):
    a = 0
    for i in range(len(q)):
        if q[i] - (i + 1) > 2:
            print("Too chaotic")
            return
        
        b = max(0, q[i] - 2)
        for j in range(b, i):
            if q[j] > q[i]:
                a += 1

    print(a)

t = int(input().strip())
for i in range(t):
    n = int(input().strip())
    q = list(map(int, input().split()))
    new_year_chaos(q)
